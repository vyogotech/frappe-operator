1. Verify operator is running:
   
   kubectl get pods -n {{ include "frappe-operator.namespace" . }}

2. Check installed CRDs:
   
   kubectl get crds | grep vyogo.tech

{{- if .Values.mariadb.enabled }}

3. Verify MariaDB is ready:
   
   kubectl get mariadb {{ .Values.mariadb.name }} -n {{ .Release.Namespace }}
   kubectl wait --for=condition=Ready mariadb/{{ .Values.mariadb.name }} -n {{ .Release.Namespace }} --timeout=300s
{{- end }}

4. Deploy your first Frappe site:

   Create a file named my-site.yaml:
   
   ---
   apiVersion: vyogo.tech/v1alpha1
   kind: FrappeBench
   metadata:
     name: my-bench
     namespace: default
   spec:
     frappeVersion: "version-15"
     apps:
       - name: erpnext
         source: image
   ---
   apiVersion: vyogo.tech/v1alpha1
   kind: FrappeSite
   metadata:
     name: my-site
     namespace: default
   spec:
     benchRef:
       name: my-bench
       namespace: default
     siteName: my-site.example.com
     dbConfig:
       provider: mariadb
       mode: shared
       mariadbRef:
         name: {{ .Values.mariadb.name }}
         namespace: {{ .Release.Namespace }}
     domain: my-site.example.com
     ingress:
       enabled: true
     ingressClassName: nginx

   Then apply:
   
   kubectl apply -f my-site.yaml

5. Monitor deployment:
   
   kubectl get frappebench,frappesite -w

6. Get auto-generated admin password:
   
   kubectl get secret my-site-admin -o jsonpath='{.data.password}' | base64 -d
